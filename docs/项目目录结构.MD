# GitTimePrism 项目目录结构

```
GitTimePrism/
├── .gitignore                    # Git 忽略规则配置
├── .trae/                        # Trae 编辑器配置目录
│   ├── rules/                    # 项目规则文件
│   │   ├── git-commit-message.md # Git 提交信息规范
│   │   └── 代码开发规范.md         # 代码开发规范
│   ├── skills/                   # 编辑器技能配置
│   └── documents/               # 内部规划文档
├── AGENTS.md                     # Agent 配置文件
├── package.json                  # Node.js 项目配置（前端依赖和脚本）
├── tsconfig.json                 # TypeScript 编译器配置
├── vite.config.js                # Vite 构建工具配置
├── index.html                    # 前端 HTML 入口文件
├── skills-lock.json              # 技能锁定文件
│
├── src/                          # 前端源代码目录（TypeScript）
│   ├── main.ts                   # 前端主入口文件（初始化应用）
│   │
│   ├── styles/                   # CSS 样式文件目录
│   │   ├── variables.css         # CSS 变量定义（暗色/亮色双主题色板）
│   │   ├── global.css            # 全局样式（CSS 重置、滚动条、按钮）
│   │   └── components.css        # 组件样式（工具栏、侧边栏、面板等）
│   │
│   ├── components/               # UI 组件目录
│   │   ├── app.ts                # 应用主框架组件（三栏布局、拖拽调整、主题切换）
│   │   ├── git-installer.ts      # Git 安装引导组件（检测 Git、下载按钮）
│   │   └── terminal.ts           # 内嵌终端组件（xterm.js + PTY 交互）
│   │
│   ├── services/                 # 业务逻辑服务目录
│   │   └── i18n.ts               # 国际化服务（语言包加载、切换、翻译）
│   │
│   ├── utils/                    # 工具函数目录
│   │   └── dom.ts                # DOM 操作工具函数
│   │
│   └── i18n/                     # 国际化语言包目录
│       ├── zh-CN.json            # 中文语言包
│       └── en-US.json            # 英文语言包
│
├── src-tauri/                    # Tauri Rust 后端目录
│   ├── Cargo.toml                # Rust 依赖配置
│   ├── Cargo.lock                # Rust 依赖锁定文件
│   ├── build.rs                  # Tauri 构建脚本
│   ├── tauri.conf.json           # Tauri 应用配置（窗口、权限、打包等）
│   ├── capabilities/             # Tauri 权限配置目录
│   │   └── default.json          # 默认权限配置（IPC 命令权限声明）
│   ├── icons/                    # 应用图标目录
│   │   ├── 32x32.png             # 32x32 像素图标
│   │   ├── 128x128.png           # 128x128 像素图标
│   │   ├── 128x128@2x.png        # 128x128 高分辨率图标
│   │   ├── icon.icns             # macOS 图标
│   │   └── icon.ico              # Windows 图标
│   │
│   └── src/                      # Rust 源代码目录
│       ├── main.rs               # 桌面端程序入口（调用 lib.rs 的 run 函数）
│       ├── lib.rs                # Rust 后端核心入口（注册命令、插件、状态）
│       ├── commands/             # Tauri IPC 命令目录（前端可调用的函数）
│       │   ├── mod.rs            # 命令模块入口
│       │   ├── system.rs         # 系统命令（Git 检测、打开外部链接）
│       │   └── terminal.rs       # 终端命令（PTY 创建、写入、调整大小、终止）
│       └── utils/                # Rust 工具函数目录
│           ├── mod.rs            # 工具模块入口
│           ├── watcher.rs        # 文件监听工具（基础框架版本）
│           └── process.rs        # 进程管理工具（静默命令执行封装）
│
└── docs/                         # 项目文档目录
    ├── 项目规划文档.MD             # 项目完整规划文档
    ├── 项目目录结构.MD             # 项目目录结构说明（本文件）
    └── agents/                   # Agent 相关文档
        ├── domain.md             # 领域模型文档
        ├── issue-tracker.md      # Issue 追踪器文档
        └── triage-labels.md      # 分拣标签文档
```
